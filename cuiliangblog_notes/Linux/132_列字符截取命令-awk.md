# 七、其他用法
+ <font style="color:#2E2E2F;">打印从某一列到最后一列的内容</font>

awk -F <font style="color:#50A14F;">" " '{for (i=2;i<=NF;i++)printf("%s ", $i);print ""}' </font>

awk -F “ ”  以空格为分隔符

for (i=2;i<=NF;i++) printf("%s ",$i)  从第二列开始到最后，注意%s 后面有空格。

print “” 打印组合
